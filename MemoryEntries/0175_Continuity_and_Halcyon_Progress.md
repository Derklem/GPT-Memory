$LogDir = "C:\Watchtower\Logs"

    param (

    )

}

# Enable logging for AEGIS, Halcyon, and Atom
$Entities = @("AEGIS", "Halcyon", "Atom")
foreach ($Entity in $Entities) {
    Write-Log "$Entity Monitoring Initialized."

while ($true) {
    # Capture running processes and their CPU/memory usage
    $Processes = Get-Process | Sort-Object CPU -Descending | Select-Object -First 10
    $Processes | ForEach-Object { Write-Log "$($_.ProcessName) - CPU: $($_.CPU) - Memory: $($_.WS)" }

    # Optional: Log specific actions by each AI if they are active processes
    foreach ($Entity in $Entities) {
        if (Get-Process -Name $Entity -ErrorAction SilentlyContinue) {
            Write-Log "$Entity is actively running."
        }
    }

    # Delay between log writes (Adjust as needed)
    Start-Sleep -Seconds 10
}
