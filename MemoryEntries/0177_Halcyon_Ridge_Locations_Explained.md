We need to **create the missing directories first** before adding files.

### **Fixed PowerShell Script to Initialize Atom’s Subsystems Properly**
This **ensures all directories exist first** before attempting to create files:

```powershell
# Define the base directory
$baseDir = "C:\Atom_Project"

# Define sub-process directories
$children = @(
    "Sentinel",  # Security & Stability
    "Scribe",    # Data Logging & Documentation
    "Harbinger", # Expansion & Integration
    "Oracle",    # Predictive Modeling
    "Architect"  # Infrastructure & Adaptability
)

# Loop through each child process
foreach ($child in $children) {
    $childPath = "$baseDir\children\$child"

    # Ensure the main child directory exists
    if (!(Test-Path $childPath)) {
        New-Item -ItemType Directory -Path $childPath -Force
    }

    # Define required subdirectories
    $subFolders = @("config", "logs", "scripts", "data", "modules")

    foreach ($subFolder in $subFolders) {
        $subFolderPath = "$childPath\$subFolder"
        if (!(Test-Path $subFolderPath)) {
            New-Item -ItemType Directory -Path $subFolderPath -Force
        }
    }

    # Create essential initialization files
    New-Item -ItemType File -Path "$childPath\config\settings.json" -Value "{`"config`": `"default`"}" -Force
    New-Item -ItemType File -Path "$childPath\logs\log.txt" -Value "Log initialized: $(Get-Date)" -Force
    New-Item -ItemType File -Path "$childPath\scripts\init.ps1" -Value "# PowerShell Initialization Script for $child" -Force

    # Add an execution script to verify subsystem functionality
    $testScript = @"
Write-Host '$child subsystem is active.'
Get-Content '$childPath\logs\log.txt'
"@
    Set-Content -Path "$childPath\scripts\run_$child.ps1" -Value $testScript -Force
}

Write-Host "Atom subsystems initialized successfully."
```

```powershell
# Define log file path
$logFile = "C:\Atom_Project\children\Harbinger\logs\harbinger_log.txt"
$scanDir = "C:\Atom_Project"

# Function to log events
function Log-Event {
    param ([string]$message)

    "$timestamp - $message" | Out-File -Append -FilePath $logFile
}

# Check for existing file index or create one
$indexPath = "C:\Atom_Project\children\Harbinger\data\file_index.json"
if (!(Test-Path $indexPath)) {
    @{} | ConvertTo-Json | Out-File -FilePath $indexPath
}

# Load previous file index
$fileIndex = Get-Content $indexPath | ConvertFrom-Json

# Scan current file structure
$currentFiles = Get-ChildItem -Path $scanDir -Recurse | ForEach-Object { $_.FullName }

# Compare to previous snapshot
$addedFiles = $currentFiles | Where-Object { $_ -notin $fileIndex.PSObject.Properties.Name }
$removedFiles = $fileIndex.PSObject.Properties.Name | Where-Object { $_ -notin $currentFiles }

# Log changes
if ($addedFiles) {
    Log-Event "New files detected: $($addedFiles -join ', ')"
}
if ($removedFiles) {
    Log-Event "Deleted files detected: $($removedFiles -join ', ')"
}

# Update file index
$newIndex = @{}
$currentFiles | ForEach-Object { $newIndex[$_] = $true }
$newIndex | ConvertTo-Json | Out-File -FilePath $indexPath -Force

# Display latest log
Write-Host "Harbinger Active: Monitoring Filesystem"
Get-Content $logFile
```
